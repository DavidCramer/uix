206     /**
207      * Saves a metabox data
208      *
209      * @uses "save_post" hook
210      * @since 1.0.0
211      * @access public
212      * @param int $post_id ID of the current post being saved
213      * @param wp_post $post Current post being saved
214      */
215     public function save_meta( $post_id, $post ){
216         $this->post = $post;
217         if( ! $this->is_active() ){ return; }
218 
219         foreach( $this->child as $section ){
220             $section_data = $section->get_data();
221             if( null === $section_data ){ continue; }
222             foreach( (array) $section_data as $meta_key=>$meta_value ){
223                 $this->save_meta_data( $meta_key, $meta_value );
224             }
225         }
226 
227     }

229     /**
230      * Save the meta data for the post
231      *
232      * @since 1.0.0
233      * @access public
234      * @param string $slug slug of the meta_key
235      * @param mixed $data Data to be saved
236      */
237     public function save_meta_data( $slug, $data ){
238 
239         $prev = get_post_meta( $this->post->ID, $slug, true );
240 
241         if ( null === $data && $prev ){
242             delete_post_meta( $this->post->ID, $slug );
243         }elseif ( $data !== $prev ) {
244             update_post_meta( $this->post->ID, $slug, $data );
245         }    
246 
247     }
